    def query_string(params)
      query = case params
        when Hash then params.map{|key,value| url_encode_param(key,value) }.join("&")
        else url_encode(params.to_s)
      end
      if !(query == nil || query.length == 0) && query[0,1] != '?'
        query = "?#{query}"
      end
      query
    end
